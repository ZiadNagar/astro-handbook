18 · Environment Variables

Environment Variables

Naming Convention

Prefix Accessible In Example
PUBLIC_ Server + Client (browser) PUBLIC_API_URL
(no prefix) Server only DATABASE_URL 
.env
DATABASE_URL=postgres://localhost/mydb     # server only
PUBLIC_API_URL=https://api.mysite.com      # exposed to browser

Accessing Variables

---
// Server-side (both work in .astro frontmatter)
const dbUrl = import.meta.env.DATABASE_URL;
const apiUrl = import.meta.env.PUBLIC_API_URL;
---

// Client-side JS (only PUBLIC_ is available)
const apiUrl = import.meta.env.PUBLIC_API_URL;

Type-Safe Env (Astro v5) NEW

astro.config.mjs
import { defineConfig, envField } from 'astro/config';

export default defineConfig({
  env: {
    schema: {
      DATABASE_URL: envField.string({ context: 'server', access: 'secret' }),
      PUBLIC_API_URL: envField.string({ context: 'client', access: 'public' }),
    }
  }
});
